#### BOJ 17609 - 회문

- 회문 판정 알고리즘 이용 
- dp 까지는 사용안해도 시간 안에 들어옴

```
#include <bits/stdc++.h>
using namespace std;

string str;
int g(int i , int j){
    if (i >= j)
        return 0;
    if (str[i] != str[j]){
        return 2;
    }
    return g(i + 1, j - 1);
}
int f(int i,int j){
    if (i >= j)
        return 0;
    if (str[i] != str[j]){
        int ret1 = g(i+1, j);
        int ret2 = g(i, j-1);
        if(ret1 == 0 || ret2 == 0) return 1;
        else return 2;
    }
    return f(i + 1, j - 1);
}

int main(void){
    int T; cin >> T;
    while(T--){
        cin >> str;
        cout << f(0, str.size()-1) << "\n";
    }
}
```

#### BOJ 17610 - 양팔저울

- 브루트포스 알고리즘 문제

```
#include <bits/stdc++.h>
using namespace std;

int N,sum, ans,S[14], c[4040404];

void f(int n, int k){
    if(n == N){
        if(k >0) c[k] =1;
        return;
    }
    f(n+1, k+S[n]);
    f(n+1, k-S[n]);
    f(n+1, k);
}

int main(void){
    cin >> N;
    
    for(int i = 0; i<N;i++) cin >> S[i];
    for(int i = 0; i<N;i++) sum += S[i];
    
    f(0,0);
    for(int i = 1;i<sum;i++){
        if(!c[i]) ans++;
    }
    cout <<ans <<"\n";
    
}
```
